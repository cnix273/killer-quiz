Activity 3: Traverse-That-DOM

Aim: Change the styling of elements on the page using JavaScript.

//Get the div within the document with an id of "articles" using "document.getElementById"
var articles = document.getElementById("articles");

//Target the div using "articles." and change the font size of all elements within the div to 50px using "style.fontSize"
articles.style.fontSize = "50px";

//Target the div's first child's last child using "articles.firstElementChild.lastElementChild" and change the font color to blue using "style.color"
articles.firstElementChild.lastElementChild.style.color = "blue";

//Target the div's previous sibling using "articles..previousElementSibling" and change the background color to black using "style.backgroundColor"
articles.previousElementSibling.style.backgroundColor = "black";

//Get the div within the document with an id of "main" using "document.getElementById"
var main = document.getElementById("main");

//Target the div's second child using "main.children[1]" and underline it using "style.textDecoration"
main.children[1].style.textDecoration = "underline";

//Target the div's last child element using "main.lastElementChild" and change the font size to 50px using "style.fontSize"
main.lastElementChild.style.fontSize = "50px";

//Target the div's first child element using "main.firstElementChild" and change the font color to orange using "style.color"
main.firstElementChild.style.color = "orange";

//Target the div's last child element's parent, which is the div itself, using "main," and change the font size to 40px using "style.fontSize"
main.style.fontSize = "40px";
